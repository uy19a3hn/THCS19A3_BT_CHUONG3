# Bài 1: Clone project từ GitHub về máy

# Bài 2: Sửa README.md rồi commit và push

git push origin main

# Bài 3: Tạo các nhánh và thêm file tương ứng

git checkout -b branch_1 main

git add text_branch_1.txt
git commit -m "Thêm text_branch_1.txt"
git push origin branch_1

git checkout main
